02 — Pydantic v2

BaseModel Basics

A Pydantic model is a Python class that defines the shape of your data. It validates, coerces, and serializes automatically. It's the backbone of all FastAPI input/output.

modelDefining your first model
Every field needs a type annotation. Pydantic coerces compatible types — e.g. the string "42" becomes 42 for an int field.
from pydantic import BaseModel
from typing import Optional
from datetime import datetime

class User(BaseModel):
    id: int
    name: str
    email: str
    age: Optional[int] = None     # optional — defaults to None
    is_active: bool = True           # optional — has a default
    created_at: datetime = datetime.utcnow()

# Create an instance — Pydantic validates on __init__
user = User(id=1, name="Alice", email="alice@example.com")

user.model_dump()       # → plain Python dict
user.model_dump_json()  # → JSON string
user.model_json_schema()# → JSON Schema dict

configModel configuration with ConfigDict
from pydantic import BaseModel, ConfigDict

class StrictUser(BaseModel):
    model_config = ConfigDict(
        str_strip_whitespace=True,  # "  Alice  " → "Alice"

        populate_by_name=True,       # allow both alias and field name
    )
    name: str
    email: str
03 — Pydantic v2

Fields & Validators

Use Field() to add constraints, metadata, and defaults. Use validators to run custom logic before or after Pydantic parses the data.

Field()Field constraints and metadata
from pydantic import BaseModel, Field

class Product(BaseModel):
    name: str  = Field(min_length=2, max_length=100)
    price: float = Field(gt=0, description="USD price, must be positive")
    quantity: int = Field(default=1, ge=0, le=9999)
    sku: str   = Field(pattern=r"^[A-Z]{2}-\d{4}$")
    tags: list[str] = Field(default_factory=list, max_length=10)

# Constraint cheatsheet:
# gt / lt      → greater / less than (exclusive)
# ge / le      → greater / less than or equal
# min_length   → min string / list length
# max_length   → max string / list length
# pattern      → regex pattern for strings
# multiple_of  → value must be a multiple of N
AnnotatedReusable constrained types (v2 best practice)
from typing import Annotated
from pydantic import Field

# Define once — use everywhere
PositiveFloat = Annotated[float, Field(gt=0)]
Username      = Annotated[str,   Field(min_length=3, max_length=30, pattern=r"^\w+$")]
PaginationInt = Annotated[int,   Field(ge=0, le=1000)]

class Order(BaseModel):
    username: Username
    total:    PositiveFloat
    skip:     PaginationInt = 0
@field_validatorField-level validation
from pydantic import field_validator

class User(BaseModel):
    email: str
    name: str

    @field_validator("email")
    @classmethod
    def check_email(cls, v):
        if "@" not in v:
            raise ValueError("bad email")
        return v.lower() # transform!

    @field_validator("name", mode="before")
    @classmethod
    def strip_name(cls, v):
        return v.strip().title()
@model_validatorCross-field validation
from pydantic import model_validator
from datetime import date

class DateRange(BaseModel):
    start: date
    end: date

    # mode="after" → typed Python objects
    @model_validator(mode="after")
    def check_range(self):
        if self.end < self.start:
            raise ValueError(
                "end must be after start")
        return self

    # mode="before" → raw dict input
    @model_validator(mode="before")
    @classmethod
    def check_keys(cls, data):
        return data
ℹ️
mode="before" receives raw input (strings, dicts). mode="after" receives fully-typed Python objects. Use "after" when comparing typed fields like dates or numbers.
04 — Pydantic v2

Nested Models & Serialization

Models can contain other models. Use aliases to control JSON key names. Use computed fields for derived values. Control output with model_dump() options.

nestedModels inside models
from pydantic import BaseModel

class Address(BaseModel):
    street: str
    city: str
    country: str = "BD"

class UserWithAddress(BaseModel):
    name: str
    addresses: list[Address]   # list of nested models
    primary: Address            # single nested model

# Pass nested data as plain dicts — Pydantic builds the models
user = UserWithAddress.model_validate({
    "name": "Rahim",
    "primary": {"street": "Mirpur 10", "city": "Dhaka"},
    "addresses": [{"street": "Gulshan 1", "city": "Dhaka"}],
})
aliasJSON key aliases
from pydantic import Field, AliasChoices

class Item(BaseModel):
    # JSON key "item_name" → Python .name
    name: str = Field(alias="item_name")

    # Accept multiple possible keys
    user_id: int = Field(
        validation_alias=AliasChoices(
            "user_id", "userId", "uid"
        )
    )

# CamelCase alias generator
from pydantic import ConfigDict
class CamelModel(BaseModel):
    model_config = ConfigDict(
        alias_generator=lambda s: ''.join(
            w.capitalize() if i else w
            for i, w in enumerate(s.split("_"))),
        populate_by_name=True,
    )
    first_name: str  # accepts "firstName" in JSON
computedComputed fields
from pydantic import BaseModel, computed_field

class Rectangle(BaseModel):
    width: float
    height: float

    @computed_field
    @property
    def area(self) -> float:
        return self.width * self.height

    @computed_field
    @property
    def perimeter(self) -> float:
        return 2 * (self.width + self.height)

r = Rectangle(width=3, height=4)
r.model_dump()
# → {'width':3,'height':4,'area':12,'perimeter':14}
model_dumpControlling serialization output
# Exclude specific fields
user.model_dump(exclude={"password", "secret_key"})

# Include ONLY specific fields
user.model_dump(include={"id", "email"})

# Use alias keys in output (camelCase etc.)
user.model_dump(by_alias=True)

# Skip None values
user.model_dump(exclude_none=True)

# Skip fields that are still at their default value
user.model_dump(exclude_defaults=True)

# Combine options
user.model_dump(exclude={"password"}, exclude_none=True, by_alias=True)
05 — FastAPI

Routes & HTTP Methods

Routes map an HTTP method + URL path to a Python function. FastAPI handles JSON parsing, type validation, and response serialization automatically.

methodsHTTP method decorators
GET → read / list data
POST → create new resource
PUT → full replacement
PATCH → partial update
DELETE → remove resource
@app.get("/items")               # list all
@app.get("/items/{item_id}")     # get one by id
@app.post("/items")              # create new
@app.put("/items/{item_id}")     # replace entirely
@app.patch("/items/{item_id}")   # partial update
@app.delete("/items/{item_id}")  # delete
CRUDComplete CRUD example
from fastapi import FastAPI, HTTPException, status
from pydantic import BaseModel

app = FastAPI()
db: dict[int, dict] = {}

class ItemCreate(BaseModel):
    name: str
    price: float

class ItemResponse(BaseModel):
    id: int
    name: str
    price: float

@app.get("/items/{item_id}", response_model=ItemResponse)
def get_item(item_id: int):
    if item_id not in db:
        raise HTTPException(status_code=404, detail="Not found")
    return db[item_id]

@app.post("/items", response_model=ItemResponse,
          status_code=status.HTTP_201_CREATED)
def create_item(item: ItemCreate):
    new_id = len(db) + 1
    db[new_id] = {"id": new_id, **item.model_dump()}
    return db[new_id]

@app.put("/items/{item_id}", response_model=ItemResponse)
def update_item(item_id: int, item: ItemCreate):
    if item_id not in db:
        raise HTTPException(status_code=404)
    db[item_id] = {"id": item_id, **item.model_dump()}
    return db[item_id]

@app.delete("/items/{item_id}", status_code=status.HTTP_204_NO_CONTENT)
def delete_item(item_id: int):
    db.pop(item_id, None)
💡
Always use response_model= on your routes. FastAPI filters the return value through it, so internal fields like passwords are automatically excluded from responses.
06 — FastAPI

Path, Query & Body Parameters

FastAPI figures out parameter source automatically: path params come from the URL, query params from ?key=val, and Pydantic models from the JSON body.

pathPath parameters — from the URL
from fastapi import Path

# Basic — auto-validated as int
@app.get("/items/{item_id}")
def get_item(item_id: int): ...

# Constrained with Path()
@app.get("/users/{user_id}")
def get_user(user_id: int = Path(ge=1, description="User ID")):
    ...

# String path param with enum validation
from enum import Enum
class ModelName(str, Enum):
    gpt4 = "gpt4"
    claude = "claude"

@app.get("/models/{model}")
def get_model(model: ModelName): ...
queryQuery parameters — from ?key=value
Any non-path, non-body parameter becomes a query param. Types are auto-coerced — "false" → False for bools.
from fastapi import Query

@app.get("/items")
def list_items(
    skip:   int       = 0,          # ?skip=0
    limit:  int       = 10,         # ?limit=10
    search: str | None = None,       # ?search=apple (optional)
    active: bool      = True,        # ?active=false
    tags:   list[str] = Query([]),  # ?tags=a&tags=b
):
    ...

# Constrained query param
@app.get("/search")
def search(q: str = Query(min_length=3, max_length=50, description="Search query")):
    ...
bodyRequest body — from JSON payload
from fastapi import Body

class CreateUser(BaseModel):
    name: str
    email: str
    password: str

# Pydantic model → JSON body
@app.post("/users")
def create_user(user: CreateUser):
    return {"id": 1, "name": user.name}

# Mix all three types in one handler
@app.put("/users/{user_id}")
def update_user(
    user_id: int,                 # ← path param
    force:   bool = False,         # ← query param
    user:    CreateUser = ...,     # ← JSON body
):
    ...
07 — FastAPI

Response Models & Status Codes

Use response_model to define what gets returned. FastAPI validates and serializes it — sensitive fields like passwords are automatically stripped out.

patternSeparate input / output schemas
# What the client SENDS (no id, no timestamps)
class UserCreate(BaseModel):
    name: str;  email: str;  password: str

# What the API RETURNS (no password!)
class UserResponse(BaseModel):
    id: int;  name: str;  email: str;  created_at: datetime

# response_model filters the return value through UserResponse
# → password is NEVER leaked, even if you return the raw DB row
@app.post("/users", response_model=UserResponse, status_code=201)
def create_user(data: UserCreate):
    user = save_to_db(data)  # full row with password hash
    return user              # password filtered by response_model
statusCommon HTTP status codes
200 OKdefault for GET
201 Createdafter POST creates resource
204 No ContentDELETE success, no body
400 Bad Requestinvalid client input
401 Unauthorizednot authenticated
403 Forbiddenno permission
404 Not Foundresource missing
422 Unprocessablevalidation error (auto)
500 Server Errorunhandled exception
errorsRaising HTTP errors
from fastapi import HTTPException, status

# Raise a 404
raise HTTPException(
    status_code=404,
    detail="User not found"
)

# Using status constants (readable)
raise HTTPException(
    status_code=status.HTTP_403_FORBIDDEN,
    detail="Not authorized",

# Custom exception handler
@app.exception_handler(ValueError)
async def val_err(req, exc):
    return JSONResponse(
        status_code=400,
        content={"detail": str(exc)}
    )
08 — FastAPI

Dependency Injection

The Depends() system lets you share logic (DB sessions, auth, pagination) across routes cleanly. A dependency is just a function FastAPI calls before your handler.

basicsHow Depends() works
from fastapi import Depends

# 1. Define a dependency — just a regular function
def get_pagination(skip: int = 0, limit: int = 10) -> dict:
    return {"skip": skip, "limit": min(limit, 100)}

# 2. Use it with Depends() — FastAPI calls it for you
@app.get("/users")
def list_users(page: dict = Depends(get_pagination)):
    # page = {"skip": 0, "limit": 10}
    return get_users(**page)

# Dependencies can depend on other dependencies (nesting)
def get_current_active_user(
    user = Depends(get_current_user),   # ← nested dep
    db   = Depends(get_db),
):
    if not user.is_active:
        raise HTTPException(400, "Inactive user")
    return user
authAuthentication dependency
from fastapi import Depends, HTTPException, Header

def get_current_user(authorization: str = Header()):
    token = authorization.removeprefix("Bearer ")
    user = verify_token(token)
    if not user:
        raise HTTPException(status_code=401, detail="Invalid token")
    return user

# Reused across many routes — DRY code
@app.get("/profile")
def profile(user = Depends(get_current_user)):
    return user

@app.delete("/account")
def delete(user = Depends(get_current_user)):
    delete_user(user.id)
yieldDB session with cleanup
def get_db():
    db = SessionLocal()
    try:
        yield db     # route runs here
    finally:
        db.close()   # always runs

@app.get("/users")
def list_users(
    db = Depends(get_db)
):
    return db.query(User).all()
router-levelApply dep to whole router
# All routes in this router
# require authentication
router = APIRouter(
    prefix="/admin",
    dependencies=[
        Depends(get_current_user),
        Depends(require_admin),
    ],
)

# Or apply to entire app
app = FastAPI(
    dependencies=[
        Depends(rate_limiter)
    ]
)
💡
Use yield in dependencies for setup/teardown patterns (DB sessions, file handles). Code before yield runs before the route; code in finally always runs after the response is sent.

Routers & Middleware

Split your routes into separate files with APIRouter. Add cross-cutting concerns like CORS, logging, and timing with middleware.

routerDefining routes in a separate file
# routers/users.py
from fastapi import APIRouter, Depends

router = APIRouter(
    prefix="/users",       # all routes start with /users
    tags=["users"],         # groups in Swagger docs

    responses={404: {"description": "Not found"}},
)

@router.get("/")             # → GET /users/
def list_users(): ...

@router.get("/{user_id}")   # → GET /users/{user_id}
def get_user(user_id: int): ...
includeMount routers in main.py
# main.py
from fastapi import FastAPI
from .routers import users, products, orders

app = FastAPI()

app.include_router(users.router)
app.include_router(products.router, prefix="/products", tags=["products"])
app.include_router(orders.router,   prefix="/v2/orders")  # versioned
corsCORS middleware
from fastapi.middleware.cors import CORSMiddleware

app.add_middleware(
    CORSMiddleware,
    allow_origins=[
        "https://myapp.com",
        "http://localhost:3000",
    ],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
customCustom middleware
import time

@app.middleware("http")
async def add_timing(request, call_next):
    start = time.time()
    response = await call_next(request)
    elapsed = time.time() - start
    response.headers["X-Process-Time"] = (
        f"{elapsed:.4f}s"
    )
    return response
10 — FastAPI

Async Routes & Lifespan

FastAPI supports sync and async handlers. Use async def when doing I/O (DB, HTTP calls). Use lifespan for startup/shutdown logic.

syncSync route handler
Runs in a thread pool. OK for blocking/sync libraries like standard SQLAlchemy.
@app.get("/items")
def get_items():
    items = db.query(Item).all()
    return items
asyncAsync route handler
Use with asyncpg, Motor, httpx, or any awaitable I/O.
@app.get("/items")
async def get_items():
    items = await db.fetch_all(
        query
    )
    return items
lifespanStartup & shutdown with lifespan
The modern replacement for the deprecated @app.on_event("startup"). Wrap setup in a context manager.
from contextlib import asynccontextmanager
from fastapi import FastAPI

@asynccontextmanager
async def lifespan(app: FastAPI):
    # ─── STARTUP ───────────────────────
    print("Starting up...")
    await connect_to_db()
    load_ml_model()

    yield  # ← app accepts requests here

    # ─── SHUTDOWN ──────────────────────
    print("Shutting down...")
    await disconnect_db()

app = FastAPI(lifespan=lifespan)
backgroundBackgroundTasks — run after response
from fastapi import BackgroundTasks

def send_welcome_email(email: str):
    # Runs AFTER the HTTP response is already returned
    send_email(to=email, subject="Welcome!")

@app.post("/users")
def create_user(user: UserCreate, bg: BackgroundTasks):
    new_user = save_user(user)
    bg.add_task(send_welcome_email, new_user.email)
    return new_user   # returned immediately, email sent after
⚠️
BackgroundTasks shares the same process. For heavy work (emails, image processing, queues), use Celery, ARQ, or Redis Queue instead.

Settings & Config

Use Pydantic's BaseSettings to load configuration from environment variables or .env files — with full type validation and IDE autocomplete.

BaseSettingsTyped config from env vars
# pip install pydantic-settings
from pydantic_settings import BaseSettings, SettingsConfigDict

class Settings(BaseSettings):
    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        case_sensitive=False,  # DB_URL == db_url
    )

    app_name:       str      = "My API"
    debug:          bool     = False
    database_url:   str               # required — no default
    secret_key:     str
    allowed_hosts:  list[str] = ["*"]
    max_connections:int      = 10
lru_cacheLoad settings once, inject anywhere
from functools import lru_cache

@lru_cache
def get_settings() -> Settings:
    return Settings()  # reads .env once, then cached forever

# Inject into any route as a dependency
from fastapi import Depends

@app.get("/info")
def app_info(settings: Settings = Depends(get_settings)):
    return {
        "app": settings.app_name,
        "debug": settings.debug,
    }

12 — Real World

Common Real-World Patterns

Patterns you'll encounter in every production FastAPI app — partial updates, polymorphism, pagination, headers, and more.

PATCHPartial updates with optional fields
class UserUpdate(BaseModel):
    # All fields optional — only provided fields are updated
    name:  str | None = None
    email: str | None = None
    age:   int | None = None

@app.patch("/users/{user_id}", response_model=UserResponse)
def patch_user(user_id: int, updates: UserUpdate):
    existing = get_user_or_404(user_id)
    updated = existing.model_copy(
        update=updates.model_dump(exclude_none=True)  # only set fields
    )
    return save(updated)
unionDiscriminated unions — polymorphic models
from typing import Annotated, Union, Literal

class Cat(BaseModel):
    type: Literal["cat"]
    indoor: bool

class Dog(BaseModel):
    type: Literal["dog"]
    breed: str

# Pydantic uses "type" field to pick the right model — fast O(1) lookup
Pet = Annotated[Union[Cat, Dog], Field(discriminator="type")]

class Owner(BaseModel):
    name: str
    pet: Pet

owner = Owner(name="Ali", pet={"type": "dog", "breed": "Labrador"})
type(owner.pet)  # → Dog
paginationGeneric paginated response
from typing import Generic, TypeVar
T = TypeVar("T")

class Page(BaseModel, Generic[T]):
    items: list[T]
    total: int
    skip:  int
    limit: int

@app.get("/users",
    response_model=Page[UserResponse])
def list_users(
    skip: int=0, limit: int=10
):
    return {
        "items": users[skip:skip+limit],
        "total": len(users),
        "skip": skip,
        "limit": limit,
    }
headersReading headers & cookies
from fastapi import Header, Cookie

@app.get("/track")
def track(
    user_agent: str | None = Header(None),
    x_request_id: str | None = Header(None),
    session_id: str | None = Cookie(None),

    return {
        "agent": user_agent,
        "rid": x_request_id,
    }
# Note: hyphen → underscore auto-conversion
# "X-Request-Id" → x_request_id
