Iterator Protocol
class Range:
    def __init__(self, n):
        self.n = n
        self.i = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.i >= self.n:
            raise StopIteration
        val = self.i
        self.i += 1
        return val

Decorators
from functools import wraps

# Basic decorator
def log(func):
    @wraps(func)
    def wrapper(*a, **kw):
        print(f"→ {func.__name__}")
        result = func(*a, **kw)
        print(f"← {result}")
        return result
    return wrapper

# Decorator factory (with args)
def retry(times=3):
    def decorator(func):
        @wraps(func)
        def wrapper(*a, **kw):
            for _ in range(times):
                try: return func(*a, **kw)
                except: pass
        return wrapper
    return decorator

asyncio — Core Concepts & Basics

07
The Mental Model async
# asyncio = single-threaded cooperative multitasking via an event loop
# "await" suspends the current coroutine and hands control back to the loop
# Only ONE coroutine runs at a time, but the loop switches between them at await points
#
#  CPU-bound work  → multiprocessing.ProcessPoolExecutor  (bypasses GIL)
#  I/O-bound work  → asyncio  (no threads needed; scales to thousands of connections)
#  Thread I/O      → asyncio.run_in_executor (integrate blocking libs)
Key rule: never call blocking code (time.sleep, requests.get, open for large files) inside a coroutine — it blocks the entire event loop. Use asyncio.sleep, aiohttp, aiofiles instead.
Coroutines & async/await

# Coroutine function → calling it returns a coroutine object

# Entry point — asyncio.run() creates + runs + closes event loop

Tasks — concurrent execution
async def main():
    # Create task → schedules coroutine immediately
    task = asyncio.create_task(fetch("url"))

    # Do other work while task runs concurrently
    await asyncio.sleep(0)  # yield to event loop

    result = await task      # get result

    # Cancel a task
    task.cancel()
    try:
        await task
    except asyncio.CancelledError:
        print("cancelled")

    # TaskGroup (3.11+) — preferred over gather
    async with asyncio.TaskGroup() as tg:
        t1 = tg.create_task(work1())
        t2 = tg.create_task(work2())
    # all tasks done here; exceptions propagate

Concurrency Patterns & Best Practices

11
Rate-limited fan-out
# Fetch many URLs, max 20 concurrent
async def fetch_all(urls: list[str]) -> list:
    sem = asyncio.Semaphore(20)

    async def bounded(url):
        async with sem:
            return await fetch(url)

    return await asyncio.gather(
        *[bounded(u) for u in urls],
        return_exceptions=True,
    )

# Chunked gather (avoid huge gather lists)
async def chunked(coros, size=50):
    results = []
    for i in range(0, len(coros), size):
        chunk = coros[i:i + size]
        results += await asyncio.gather(*chunk)
    return results
Retry with Backoff
import asyncio, math

async def retry(
    coro_fn,
    *args,
    retries=3,
    base_delay=1.0,
    jitter=0.3,
):
    for attempt in range(retries):
        try:
            return await coro_fn(*args)
        except (OSError, TimeoutError) as e:
            if attempt == retries - 1:
                raise
            delay = base_delay * (2 ** attempt)
            await asyncio.sleep(delay + jitter)

# Usage
result = await retry(fetch, "https://...")
Background Task Lifecycle
class BackgroundWorker:
    def __init__(self):
        self._task = None

    async def start(self):
        self._task = asyncio.create_task(
            self._run()
        )

    async def stop(self):
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass

    async def _run(self):
        try:
            while True:
                await do_work()
                await asyncio.sleep(1)
        except asyncio.CancelledError:
            await cleanup()
            raise  # always re-raise!

⚠ Don't Do This

# Fix: result = await fetch("url")

    time.sleep(2)          # freezes everything

    return requests.get(url)  # blocking!
# Fix: use aiohttp, httpx

# WRONG: bare asyncio.get_event_loop()

# Fix: asyncio.get_running_loop()
#   or  asyncio.run(main())

# WRONG: not re-raising CancelledError
async def bad():
    try: ...
    except asyncio.CancelledError:
        pass  # swallowed! task won't cancel
# Fix: always raise after cleanup

Notes
•  asyncio is single-threaded — never share mutable state between coroutines without a Lock.
•  asyncio.TaskGroup (3.11+) is the modern way to run concurrent tasks — it handles cancellation correctly.
•  "await" only works inside "async def" functions. Async code must stay async all the way to asyncio.run().
•  For true parallelism (CPU-bound): use multiprocessing or ProcessPoolExecutor. asyncio only helps I/O.
•  Libraries: aiohttp (HTTP), aiofiles (file I/O), asyncpg (PostgreSQL), aiobotocore (AWS), anyio (compatible asyncio/trio).
•  Python version: features marked 3.10+ include match/case; 3.11+ includes TaskGroup, ExceptionGroup, asyncio.timeout().
