01 — Overview

Why LangGraph?

LangChain handles model + tool integrations. LangGraph handles the orchestration — the hard part of making agents reliable at scale.

◆
Core insight: LangGraph models agent workflows as directed graphs with typed state. Every step is a node. Every transition is an edge. Complex loops, retries, branching, and human pauses are first-class primitives — not afterthoughts.
🔁
Cycles & Loops
Unlike DAG-only frameworks, LangGraph supports cycles natively. Agents can retry, reflect, loop back, and converge — essential for real reasoning patterns.
Core Differentiator
📦
Typed State
A shared TypedDict flows through every node. No implicit side effects. State is inspectable, snapshotted, and resumable at any checkpoint.
State
🛑
Interrupts
Pause execution at any node, hand off to a human, receive input, and resume — all without losing state. The foundation of safe agentic systems.
Human-in-the-Loop
♻️
Durable Execution
Agents that survive failures. LangGraph persists state at every step — if a node crashes, resume from the last checkpoint, not from scratch.
Execution
⚡
First-Class Streaming
Stream tokens, node updates, and state diffs in real time. Multiple stream modes: values, updates, messages, debug. Works out of the box.
Streaming
🧠
Memory Store
Short-term working memory (thread-scoped) and long-term cross-session memory via LangGraph Store. Agents that remember across conversations.
Memory

03 — Quick Start

From Zero to Agent

Build your first LangGraph agent. A ReAct loop with tool-calling in under 40 lines.

01

Install LangGraph

LangGraph has zero mandatory dependencies. Add your model provider separately.

02

Define your State schema

State is a TypedDict that flows through every node. MessagesState is the built-in shorthand for chat agents.

03

Write node functions

Each node is a plain Python function: takes State, returns a partial update dict.

04

Wire edges & compile

Connect nodes with edges. Add conditional edges for branching. Call .compile() to get a runnable graph.

python — minimal hello world
from langgraph.graph import StateGraph, MessagesState, START, END

# Node: any function that takes State and returns partial update
def mock_llm(state: MessagesState):
    return {"messages": [{"role": "ai", "content": "hello world"}]}

# Build graph
graph = StateGraph(MessagesState)
graph.add_node(mock_llm)
graph.add_edge(START, "mock_llm")
graph.add_edge("mock_llm", END)

# Compile → runnable
app = graph.compile()

result = app.invoke({
    "messages": [{"role": "user", "content": "hi!"}]
})
print(result["messages"][-1].content)  # hello world
python — ReAct agent with tools
from langgraph.graph import StateGraph, MessagesState, START, END
from langgraph.prebuilt import ToolNode, tools_condition
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool

# Define tools
@tool
def search(query: str) -> str:
    """Search the web for a query."""
    return f"Results for: {query}"

tools = [search]
llm = ChatOpenAI(model="gpt-4o").bind_tools(tools)

# Nodes
def call_model(state: MessagesState):
    return {"messages": [llm.invoke(state["messages"])]}

# Build
g = StateGraph(MessagesState)
g.add_node("agent", call_model)
g.add_node("tools", ToolNode(tools))

g.add_edge(START, "agent")
# Conditional: if tool calls → tools node, else → END
g.add_conditional_edges("agent", tools_condition)
g.add_edge("tools", "agent")  # loop back!

app = g.compile()
result = app.invoke({
    "messages": [{"role": "user", "content": "Who won the 2024 election?"}]
})
python — conditional edges & branching
from typing import TypedDict, Literal

class State(TypedDict):
    question: str
    topic: Literal["tech", "legal", "general"]
    answer: str

def classify(state: State) -> State:
    # Determine topic via LLM classifier
    topic = llm_classify(state["question"])
    return {"topic": topic}

# Router function — returns NEXT NODE NAME
def route(state: State) -> str:
    return state["topic"]  # "tech" | "legal" | "general"

g = StateGraph(State)
g.add_node("classify", classify)
g.add_node("tech", tech_agent)
g.add_node("legal", legal_agent)
g.add_node("general", general_agent)

g.add_edge(START, "classify")
g.add_conditional_edges(
    "classify", route,
    {"tech": "tech", "legal": "legal", "general": "general"}
)
[g.add_edge(n, END) for n in ["tech", "legal", "general"]]
python — streaming modes
# Stream full state updates after each node
for event in app.stream(inputs, stream_mode="updates"):
    for node, update in event.items():
        print(f"[{node}] → {update}")

# Stream LLM tokens as they arrive (stream_mode="messages")
async for msg, meta in app.astream(inputs, stream_mode="messages"):
    if msg.content:
        print(msg.content, end="", flush=True)

# Multiple stream modes simultaneously
async for chunk in app.astream(
    inputs,
    stream_mode=["updates", "messages", "debug"]
):
    mode, data = chunk
    print(f"{mode}: {data}")

# FastAPI SSE endpoint
from fastapi.responses import StreamingResponse

@app.post("/chat")
async def chat_stream(query: str):
    async def gen():
        async for msg, _ in graph.astream(
            {"messages": [{"role": "user", "content": query}]},
            stream_mode="messages"
        ):
            yield f"data: {msg.content}\n\n"
    return StreamingResponse(gen(), media_type="text/event-stream")
04 — State

State Management

State is the backbone of every LangGraph app. It flows through every node and is persisted at every checkpoint.

python — state schemas
from typing import TypedDict, List, Optional

# Custom state schema
class AgentState(TypedDict):
    messages: List[dict]       # conversation history
    documents: List[str]      # retrieved docs
    generation: Optional[str] # current answer draft
    retries: int               # retry counter
    route: str                 # current route decision

# Nodes return PARTIAL updates — only changed keys
def retrieve(state: AgentState):
    docs = vectorstore.similarity_search(
        state["messages"][-1]["content"]
    )
    return {"documents": docs}  # only update "documents"

def generate(state: AgentState):
    answer = llm.invoke(...)
    return {"generation": answer}  # only update "generation"
python — reducers for merge logic
from typing import Annotated
from langgraph.graph.message import add_messages
import operator

class State(TypedDict):
    # add_messages: appends new msgs instead of overwriting
    messages: Annotated[list, add_messages]

    # operator.add: list accumulation across parallel nodes
    retrieved_docs: Annotated[list, operator.add]

    # last_value: simple overwrite (default behavior)
    answer: str

# Why reducers matter in parallel branches:
# Branch A returns {"retrieved_docs": [doc1, doc2]}
# Branch B returns {"retrieved_docs": [doc3, doc4]}
# Reducer merges → [doc1, doc2, doc3, doc4] ✓
# Without reducer → last write wins, data lost ✗
python — checkpointing & persistence
from langgraph.checkpoint.memory import MemorySaver
from langgraph.checkpoint.postgres import PostgresSaver  # production

# Dev: in-memory

# Production: Postgres

    "postgresql://user:pass@host/db"
)

# Thread ID = conversation session

# Each invoke auto-saves state at every step
app.invoke(input_1, config)  # checkpoint saved
app.invoke(input_2, config)  # resumes from saved state

# Inspect any checkpoint
state = app.get_state(config)
history = list(app.get_state_history(config))  # all checkpoints

python — all edge types
g = StateGraph(State)

# ── NODES ──
g.add_node("my_node", my_function)         # named node
g.add_node(my_function)                    # auto-named from function

# ── BASIC EDGE ── always follows
g.add_edge("node_a", "node_b")

# ── CONDITIONAL EDGE ── router function decides next node
def router(state) -> str:
    if state["score"] > 0.8: return "generate"
    else: return "rewrite"

g.add_conditional_edges(
    "grade", router,
    {"generate": "generate", "rewrite": "rewrite"}
)

# ── ENTRY POINT ──
g.set_entry_point("my_node")  # or add_edge(START, "my_node")

# ── FINISH POINT ──
g.set_finish_point("my_node")  # or add_edge("my_node", END)

# ── PARALLEL BRANCHES ── fan out to multiple nodes simultaneously
g.add_edge("split", ["branch_a", "branch_b"])  # both run in parallel
06 — Architecture Patterns

Production Patterns

The canonical graph shapes for common agentic use cases. Pick the right one for your task.

⟳ ReAct Loop

Agent reasons, calls tools, observes results, repeats. The universal agent pattern. Terminates when no tool calls are made.

START → agent [if tools?] → tools → agent
[if done?] → END

🌿 Router / Classifier

Classify query, route to specialized sub-agents. Each expert handles its domain. Perfect for multi-domain systems.

START → classify
→ tech_agent | → legal_agent | → general_agent
[all] → END

🔍 Corrective RAG (CRAG)

Retrieve → grade docs → if relevant: generate. If not: rewrite query or web search. Self-healing retrieval loop.

retrieve → grade [relevant?]
→ generate → END
[irrelevant?] → rewrite → retrieve

⚡ Parallel Map-Reduce

Fan out to N parallel workers, collect results, reduce into final answer. Ideal for multi-document analysis.

START → split
→ worker_1 || → worker_2 || → worker_3
[barrier] → reduce → END

🤖 Multi-Agent Supervisor

Supervisor LLM assigns tasks to specialized agents. Each agent reports back. Supervisor decides next assignment or exits.

supervisor [assigns]
→ researcher | → coder | → writer
[each reports] → supervisor

🔗 Plan-and-Execute

Planner creates a task list upfront. Executor runs each step sequentially. Re-planner adjusts when steps fail or surface new info.

planner → executor [done?]
→ END
[needs replan?] → re-planner → executor
🧩
Subgraphs: Any compiled graph can be used as a node inside another graph. Build complex hierarchical agents by composing smaller graphs. Parent and child share state via overlapping keys.

python — interrupt_before node
from langgraph.graph import StateGraph, END, START

# Compile with interrupt_before to pause BEFORE a node executes
app = graph.compile(
    checkpointer=checkpointer,
    interrupt_before=["execute_action"]  # pause here for approval
)

config = {"configurable": {"thread_id": "thread-1"}}

# Run until interrupt
state = app.invoke(inputs, config)
# → execution stops, state is saved at checkpoint

# Inspect pending state
current = app.get_state(config)
print("Pending action:", current.values["pending_action"])

# Resume (human approved → pass None)
app.invoke(None, config)

# Or reject + override state before resuming
app.update_state(config, {"pending_action": safe_alternative})
app.invoke(None, config)
python — review tool calls before execution
from langchain_core.messages import AIMessage

# Compile with interrupt BEFORE tool execution
app = graph.compile(
    checkpointer=checkpointer,
    interrupt_before=["tools"]
)

config = {"configurable": {"thread_id": "review-1"}}
app.invoke(user_input, config)

# Check what tool calls are pending
state = app.get_state(config)
last_msg = state.values["messages"][-1]
print("Tool calls:", last_msg.tool_calls)

# EDIT: modify the tool call before execution
edited_msg = AIMessage(
    content=last_msg.content,
    tool_calls=[{
        **last_msg.tool_calls[0],
        "args": {"query": "safer search term"}
    }]
)
app.update_state(config, {"messages": [edited_msg]})
app.invoke(None, config)  # resume with edited call
python — dynamic interrupts inside nodes
from langgraph.types import interrupt

# Interrupt conditionally within a node function
def sensitive_action(state):
    action = state["proposed_action"]

    if action["risk_level"] > 0.7:
        # Pause and request human input
        human_response = interrupt({
            "question": "Approve this action?",
            "action": action,
            "risk": action["risk_level"],
        })
        # human_response contains what was passed to .invoke(None, config)
        if not human_response.get("approved"):
            return {"status": "rejected"}

    # Proceed with action
    result = execute(action)
    return {"result": result}
08 — Memory

Comprehensive Memory

Two memory systems: short-term working memory per thread, and long-term cross-session memory via the Memory Store.

Short-Term

Thread-Scoped Memory

Messages and state within a conversation. Persisted via checkpointer. Scoped to a thread_id. Automatically available for the entire session.

storage: checkpointer
scope: thread_id
lifetime: conversation
Long-Term

Memory Store

Cross-session, cross-thread storage. Use for user preferences, past interactions, and domain knowledge that should persist forever.

storage: InMemoryStore / Redis
scope: user_id (any key)
lifetime: persistent
python — long-term memory with Memory Store
from langgraph.store.memory import InMemoryStore
from langgraph.graph import StateGraph

store = InMemoryStore()  # swap for RedisStore in production

# Inject store into graph
app = graph.compile(checkpointer=checkpointer, store=store)

# Inside a node — access via RunnableConfig or direct injection
from langchain_core.runnables.config import RunnableConfig

def personalized_agent(state, config: RunnableConfig, store):
    user_id = config["configurable"]["user_id"]

    # Load user profile from long-term store
    memories = store.search((user_id, "preferences"))

    # Use memories in the prompt
    context = "\n".join(m.value["text"] for m in memories)
    response = llm.invoke(f"User context: {context}\n\n{state['messages'][-1]}")

    # Save new memory
    store.put((user_id, "preferences"), "pref-1", {"text": "Prefers concise answers"})
    return {"messages": [response]}

10 — Resilience

Durable Execution

LangGraph checkpoints state after every node. If a step fails, resume from the last checkpoint. No lost work, no full reruns.

⚠️
Without durable execution: A 20-step agent that fails on step 19 restarts from scratch. With LangGraph checkpointing, it resumes from step 18. For long-running agents (minutes to hours), this is non-negotiable.
python — production persistence with PostgreSQL
from langgraph.checkpoint.postgres import PostgresSaver
from psycopg_pool import ConnectionPool

# Production: use connection pool
pool = ConnectionPool(
    conninfo="postgresql://user:pass@db:5432/langgraph",
    max_size=20,
)
checkpointer = PostgresSaver(pool)
checkpointer.setup()  # creates tables if not exists

app = graph.compile(checkpointer=checkpointer)

# Retry from checkpoint after failure
config = {"configurable": {"thread_id": "job-789"}}

try:
    result = app.invoke(inputs, config)
except Exception as e:
    print(f"Failed: {e}. Resuming from checkpoint…")
    # Simply invoke again with same config — LangGraph
    # auto-resumes from last saved checkpoint
    result = app.invoke(None, config)

# Time-travel: revert to any previous state
history = list(app.get_state_history(config))
target_checkpoint = history[3]  # go back 3 steps
app.invoke(None, target_checkpoint.config)

12 — Production

Production Checklist

Before you ship a LangGraph agent to production. Click items to track progress.

⚙️ Graph Design

0 / 5
  • ✓Define explicit TypedDict State — no implicit side effects between nodes
  • ✓Use Annotated reducers for any key written by parallel branches
  • ✓Add recursion_limit to compile() to prevent infinite loops in production
  • ✓Ensure all conditional edge routers return valid node names (never None)
  • ✓Draw and review graph with graph.get_graph().draw_mermaid_png() before shipping

🔒 Persistence & Reliability

0 / 5
  • ✓Use PostgresSaver (not MemorySaver) for all production workloads
  • ✓Set up Postgres connection pool — not bare connections
  • ✓Test checkpoint resume: kill a running agent mid-execution, verify it resumes correctly
  • ✓Use unique, stable thread_ids (not random UUIDs) so users can resume conversations
  • ✓Set up checkpoint cleanup policy — old threads should expire or be pruned

🧑‍💻 Human-in-the-Loop

0 / 4
  • ✓Identify all high-risk actions and add interrupt_before or dynamic interrupt()
  • ✓Build a UI or webhook that presents pending state to human reviewers
  • ✓Test the update_state path — humans should be able to edit, not just approve/reject
  • ✓Set a timeout on interrupted states — auto-reject if no human response after N hours

📊 Observability & Eval

0 / 5
  • ✓Enable LangSmith tracing: LANGSMITH_TRACING=true + API key
  • ✓Create golden eval dataset with ≥50 representative Q&A pairs
  • ✓Run evals on every deploy — gate merges on no quality regression
  • ✓Add structured logging of node latency and token counts per node
  • ✓Monitor for stuck threads (long-running, no progress) in Postgres

🚀 Deployment

0 / 4
  • ✓Deploy via LangSmith Deployment or expose via FastAPI with SSE streaming
  • ✓Set max_concurrency per graph to prevent database connection exhaustion
  • ✓Implement API rate limits and per-user thread quotas
  • ✓Load test with realistic concurrency — LangGraph graphs are thread-safe, but your tools may not be

python — production FastAPI server
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import json

server = FastAPI(title="LangGraph Agent API")

class ChatRequest(BaseModel):
    message: str
    thread_id: str

@server.post("/chat/stream")
async def chat_stream(req: ChatRequest):
    config = {"configurable": {"thread_id": req.thread_id}}
    inputs = {"messages": [{"role": "user", "content": req.message}]}

    async def generate():
        async for msg, meta in app.astream(
            inputs, config, stream_mode="messages"
        ):
            if msg.content and not isinstance(msg.content, list):
                yield f"data: {json.dumps({'text': msg.content})}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")

@server.get("/state/{thread_id}")
async def get_state(thread_id: str):
    config = {"configurable": {"thread_id": thread_id}}
    return app.get_state(config).values
python — retry with backoff + max retries
from typing import TypedDict

class State(TypedDict):
    messages: list
    retries: int
    error: str

MAX_RETRIES = 3

def should_retry(state: State) -> str:
    if state.get("error") and state["retries"] < MAX_RETRIES:
        return "retry"
    elif state.get("error"):
        return "fail"
    return "success"

def handle_retry(state: State) -> State:
    import time
    time.sleep(2 ** state["retries"])  # exponential backoff
    return {"retries": state["retries"] + 1, "error": None}

g.add_conditional_edges(
    "execute", should_retry,
    {"retry": "handle_retry", "fail": "handle_fail", "success": END}
)
g.add_edge("handle_retry", "execute")

from typing import Annotated
import operator

class State(TypedDict):
    docs: list[str]
    summaries: Annotated[list[str], operator.add]  # merge from parallel
    final: str

def summarize_doc(state: State) -> State:
    # Each parallel branch gets same state but different doc
    summary = llm.invoke(f"Summarize: {state['current_doc']}")
    return {"summaries": [summary.content]}

def reduce(state: State) -> State:
    combined = "\n".join(state["summaries"])
    final = llm.invoke(f"Synthesize these summaries: {combined}")
    return {"final": final.content}

# Fan out — all branches run in parallel
