Tenants + Users

-- Enable UUID + RLS
CREATE EXTENSION IF NOT EXISTS "pgcrypto";
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

CREATE TYPE tenant_plan   AS ENUM ('free', 'pro', 'enterprise');
CREATE TYPE tenant_status AS ENUM ('trial', 'active', 'suspended');
CREATE TYPE user_status   AS ENUM ('active', 'inactive', 'banned');

CREATE TABLE tenants (
  id          UUID            PRIMARY KEY DEFAULT gen_random_uuid(),
  slug        TEXT            UNIQUE NOT NULL,
  name        TEXT            NOT NULL,
  plan        tenant_plan     NOT NULL DEFAULT 'free',
  status      tenant_status   NOT NULL DEFAULT 'trial',
  domain      TEXT,
  settings    JSONB           NOT NULL DEFAULT '{}',
  max_seats   INT             NOT NULL DEFAULT 5,
  created_at  TIMESTAMPTZ     NOT NULL DEFAULT now(),
  updated_at  TIMESTAMPTZ     NOT NULL DEFAULT now(),
  deleted_at  TIMESTAMPTZ
);
CREATE INDEX idx_tenants_domain ON tenants(domain)
  WHERE domain IS NOT NULL;

CREATE TABLE users (
  id               UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id        UUID         NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  email            TEXT         NOT NULL,
  password_hash    TEXT,        -- NULL for pure-OAuth users
  display_name     TEXT,
  avatar_url       TEXT,
  email_verified   BOOL         NOT NULL DEFAULT FALSE,
  mfa_enabled      BOOL         NOT NULL DEFAULT FALSE,
  mfa_secret       TEXT,        -- encrypted TOTP secret
  last_login_at    TIMESTAMPTZ,
  status           user_status  NOT NULL DEFAULT 'active',
  created_at       TIMESTAMPTZ  NOT NULL DEFAULT now(),
  deleted_at       TIMESTAMPTZ,
  UNIQUE(tenant_id, email)    -- email unique per tenant
);
CREATE INDEX idx_users_tenant ON users(tenant_id);

OAuth2 Tables

CREATE TYPE oauth_provider_type AS ENUM (
  'google', 'github', 'microsoft', 'okta', 'custom_oidc'
);

CREATE TABLE oauth_providers (
  id             UUID               PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id      UUID               NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
  provider       oauth_provider_type NOT NULL,
  client_id      TEXT               NOT NULL,
  client_secret  TEXT               NOT NULL, -- encrypted
  scopes         TEXT[]             NOT NULL DEFAULT '{openid,email,profile}',
  redirect_uri   TEXT               NOT NULL,
  discovery_url  TEXT,              -- OIDC /.well-known/openid-configuration
  enabled        BOOL               NOT NULL DEFAULT TRUE,
  extra_config   JSONB,
  UNIQUE(tenant_id, provider)
);

CREATE TABLE oauth_connections (
  id                UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id           UUID        NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  provider_id       UUID        NOT NULL REFERENCES oauth_providers(id),
  provider_user_id  TEXT        NOT NULL,
  access_token      TEXT        NOT NULL, -- encrypted
  refresh_token     TEXT,                  -- encrypted
  token_expires_at  TIMESTAMPTZ,
  id_token_claims   JSONB,
  connected_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(user_id, provider_id),
  UNIQUE(provider_id, provider_user_id)
);

Sessions + JWT Blocklist

CREATE TABLE sessions (
  id                   UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id              UUID        NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  tenant_id            UUID        NOT NULL REFERENCES tenants(id),
  refresh_token_hash   TEXT        UNIQUE NOT NULL,
  jwt_jti              UUID        UNIQUE NOT NULL,
  ip_address           INET,
  user_agent           TEXT,
  device_fingerprint   TEXT,
  expires_at           TIMESTAMPTZ NOT NULL,
  revoked_at           TIMESTAMPTZ,
  created_at           TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX idx_sessions_user     ON sessions(user_id);
CREATE INDEX idx_sessions_expires  ON sessions(expires_at)
  WHERE revoked_at IS NULL;

CREATE TABLE token_blocklist (
  jti         UUID        PRIMARY KEY,
  user_id     UUID        NOT NULL REFERENCES users(id),
  revoked_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  expires_at  TIMESTAMPTZ NOT NULL,
  reason      TEXT
);
-- Auto-purge expired blocklist entries
CREATE INDEX idx_blocklist_expires ON token_blocklist(expires_at);

RBAC Tables

CREATE TABLE roles (
  id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id   UUID        REFERENCES tenants(id) ON DELETE CASCADE,
  name        TEXT        NOT NULL,
  description TEXT,
  is_system   BOOL        NOT NULL DEFAULT FALSE,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(tenant_id, name)
); -- tenant_id NULL = global system role

CREATE TABLE permissions (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  action      TEXT NOT NULL,  -- create|read|update|delete|admin
  resource    TEXT NOT NULL,  -- users|projects|billing|settings …
  description TEXT,
  UNIQUE(action, resource)
);

CREATE TABLE role_permissions (
  role_id       UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
  permission_id UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
  granted_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  granted_by    UUID REFERENCES users(id),
  PRIMARY KEY(role_id, permission_id)
);

CREATE TABLE user_roles (
  user_id      UUID        NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  role_id      UUID        NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
  tenant_id    UUID        NOT NULL REFERENCES tenants(id),
  assigned_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  assigned_by  UUID        REFERENCES users(id),
  expires_at   TIMESTAMPTZ,
  PRIMARY KEY(user_id, role_id, tenant_id)
);

Audit Log (Partitioned)

CREATE TABLE audit_logs (
  id            BIGSERIAL,
  tenant_id     UUID        NOT NULL REFERENCES tenants(id),
  actor_id      UUID        REFERENCES users(id),
  action        TEXT        NOT NULL,
  resource_type TEXT        NOT NULL,
  resource_id   UUID,
  changes       JSONB,      -- {before: {}, after: {}}
  ip_address    INET,
  metadata      JSONB,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY(id, created_at)
) PARTITION BY RANGE (created_at);

-- Create monthly partitions (automate with pg_partman)
CREATE TABLE audit_logs_2025_01
  PARTITION OF audit_logs
  FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');

CREATE INDEX idx_audit_tenant    ON audit_logs(tenant_id, created_at DESC);
CREATE INDEX idx_audit_actor     ON audit_logs(actor_id, created_at DESC);
CREATE INDEX idx_audit_resource  ON audit_logs(resource_type, resource_id);

① Password + JWT Login

01
POST /auth/login
Receive email + password. Resolve tenant from subdomain or body.
02
Verify Credentials
SELECT user WHERE tenant_id + email. Compare bcrypt hash.
03
MFA Check
If mfa_enabled, verify TOTP code against mfa_secret.
04
Issue Tokens
Sign short-lived access JWT (15m) + long-lived refresh token. Store session row.
05
Return
access_token in body; refresh_token in HttpOnly Secure cookie.

② OAuth2 / OIDC Login (e.g. Google)

01
GET /auth/oauth/{provider}
Load oauth_providers row for tenant. Build authorization URL with PKCE + state param.
02
Redirect to IdP
User authenticates at Google/GitHub. state stored in session cookie.
03
Callback /auth/oauth/cb
Verify state. Exchange code → access_token + id_token via PKCE verifier.
04
Upsert User
Find or create user by email+tenant. Upsert oauth_connections with encrypted tokens.
05
Issue JWT
Same session/JWT flow as password login. Redirect to dashboard.

③ Token Refresh Flow

01
POST /auth/refresh
Read refresh_token from HttpOnly cookie.
02
Validate Session
Hash token → lookup sessions.refresh_token_hash. Check not revoked, not expired.
03
Rotate Refresh
Invalidate old refresh token. Issue new refresh token (rotation). Update session row.
04
New Access JWT
Issue new access token with fresh jti. Return in body.

④ Request Authorization (Middleware)

01
Extract JWT
Parse Bearer token from Authorization header.
02
Verify Signature
Verify RS256/ES256 signature. Check exp, iss, aud claims.
03
Blocklist Check
Query token_blocklist WHERE jti = claims.jti (Redis cache first).
04
Set RLS Context
SET LOCAL app.tenant_id = '…'; SET LOCAL app.user_id = '…' for row-level security.
05
RBAC Check
Query user_roles → role_permissions. Verify action+resource against required permission.

Tenant Isolation Policy

Every tenant-scoped table uses RLS to ensure rows are only visible to users in the correct tenant. Set via transaction-level config.

ALTER TABLE users
  ENABLE ROW LEVEL SECURITY;

CREATE POLICY tenant_isolation
  ON users
  USING (
    tenant_id = current_setting(
      'app.tenant_id'
    )::uuid
  );

Setting RLS Context in App

The app server sets tenant and user context at the start of every transaction after JWT validation.

-- In your DB transaction:
SET LOCAL app.tenant_id
  = 'ten_abc123';

SET LOCAL app.user_id
  = 'usr_xyz789';

SET LOCAL app.user_role
  = 'admin';

Audit Log Write Policy

Audit logs can only be inserted by the app role. No user can update or delete audit records, ensuring tamper-proof trail.

CREATE POLICY audit_insert
  ON audit_logs
  AS PERMISSIVE
  FOR INSERT
  TO app_role
  WITH CHECK (TRUE);

-- No UPDATE or DELETE policy
-- = nobody can modify logs

Soft Delete Filter

Global filter policy hides soft-deleted users and tenants from all queries automatically.

CREATE POLICY hide_deleted
  ON users
  USING (
    deleted_at IS NULL
  );

-- Same for tenants:
CREATE POLICY hide_deleted
  ON tenants
  USING (deleted_at IS NULL);

Self-Only User Update

Users can only update their own profile row. Admins bypass this via a separate policy or the service role.

CREATE POLICY self_update
  ON users
  FOR UPDATE
  USING (
    id = current_setting(
      'app.user_id'
    )::uuid
  );

Admin Bypass via Role

Tenant admins and system roles bypass tenant isolation. Use a separate DB role or a permissive policy combined with role check.

CREATE POLICY admin_bypass
  ON users
  AS PERMISSIVE
  USING (
    current_setting('app.user_role')
      = 'admin'
    AND tenant_id = current_setting(
      'app.tenant_id')::uuid
  );

Tenant Isolation Strategy

This schema uses shared database, shared schema with Row-Level Security — the most cost-effective model for SaaS. Every tenant-scoped table carries a tenant_id FK and a corresponding RLS policy.

The alternative is schema-per-tenant (PostgreSQL schemas) which gives stronger isolation but complicates migrations. For most SaaS products, RLS + application-level enforcement is sufficient.

Key rules: Never query without setting app.tenant_id. All API endpoints resolve tenant from JWT claim, subdomain, or custom header before touching the DB.

JWT Design

Access tokens are short-lived (15 min) signed with RS256 (asymmetric). They carry: sub (user_id), tid (tenant_id), roles array, jti (unique ID for revocation), exp.

Refresh tokens are opaque random strings stored as SHA-256 hashes. They live in HttpOnly Secure cookies, rotate on every use, and are invalidated server-side.

The token_blocklist table handles forced logout. For scale, mirror it to Redis with a TTL matching exp.

OAuth2 / OIDC Architecture

Each tenant can configure their own IdP (Google Workspace SSO, GitHub org, Okta, etc.) via oauth_providers. This enables enterprise tenants to enforce their own identity policies.

The authorization code flow uses PKCE (Proof Key for Code Exchange) to prevent interception attacks. The state parameter is a signed JWT containing the tenant context, preventing CSRF.

Tokens from the IdP are AES-256 encrypted at rest in oauth_connections. Decryption happens only when the app needs to call the IdP's API on behalf of the user.

RBAC Model

Classic Role → Permissions model. Roles can be system-wide (e.g. super_admin) or tenant-scoped (e.g. billing_admin). Permissions are action:resource pairs like read:invoices or delete:users.

Roles are embedded in the JWT as an array for fast in-process checks, but always re-verified from DB for sensitive operations. The DB is the source of truth.

Role assignments support expires_at for temporary privilege escalation (on-call access, contractor access).

Audit Log Design

Append-only, range-partitioned by month. No UPDATE or DELETE is possible via RLS policy. Use pg_partman to auto-create future partitions.

The changes JSONB column stores {before: {...}, after: {...}} diffs. Sensitive fields (passwords, tokens) are never stored.

For high write volume: route audit inserts through a message queue (Kafka / SQS) and write asynchronously to avoid adding latency to user requests.

Security Checklist

✅  bcrypt/argon2 for passwords
✅  AES-256 for stored OAuth tokens
✅  PKCE on all OAuth flows
✅  Refresh token rotation (detect reuse)
✅  HttpOnly + Secure + SameSite cookies
✅  RLS enforced at DB layer
✅  TOTP MFA stored encrypted
✅  Soft deletes (GDPR erasure via column clear)
✅  Audit log tamper-proof via RLS
✅  API keys: prefix shown, hash stored
✅  Rate limiting on auth endpoints
✅  Token blocklist auto-purge
